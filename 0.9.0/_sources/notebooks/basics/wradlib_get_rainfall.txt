The conversion is carried out the same way as in the example above. The plot is produced using\n",
      "the function [wradlib.vis.plot_ppi](http://wradlib.org/wradlib-docs/$TAG/generated/wradlib.vis.plot_ppi.html)."

      "def read_data(dtimes):\n",
      "    \"\"\"Helper function to read raw data for a list of datetimes <dtimes>\n",
      "    \"\"\"\n",
      "    data = np.empty((len(dtimes),360,128))\n",
      "    for i, dtime in enumerate(dtimes):\n",
      "        f = wrl.util.get_wradlib_data_file('dx/raa00-dx_10908-{0}-fbg---bin.gz'.format(dtime))\n",
      "        data[i], attrs = wrl.io.readDX(f)\n",
      "    return data"
